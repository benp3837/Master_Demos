First BigQuery Demo Queries:

(NYC Citibike)

-- Initial query (stations ordered by number of trips descending)
SELECT
  start_station_name,
  start_station_latitude,
  start_station_longitude,
  ST_GEOGPOINT(start_station_longitude, start_station_latitude) AS geo_location,
  COUNT(*) AS num_trips
FROM
  `bigquery-public-data.new_york.citibike_trips`
GROUP BY -- 1, 2, 3 lets us avoid writing the names of the columns in the select
  1,
  2,
  3
ORDER BY
  num_trips DESC
LIMIT
  25;

-- Trips by Customer Type
SELECT
  usertype,
  COUNT(*) AS num_trips,
  ROUND(AVG(tripduration) / 60, 2) AS avg_trip_minutes
FROM
  `bigquery-public-data.new_york.citibike_trips`
GROUP BY
  usertype
ORDER BY
  num_trips DESC;

-- Find busiest months (in 2016)
SELECT
  EXTRACT(MONTH FROM starttime) AS month,
  COUNT(*) AS num_trips
FROM
  `bigquery-public-data.new_york.citibike_trips`
WHERE
  EXTRACT(YEAR FROM starttime) = 2016
GROUP BY
  month
ORDER BY
  month ASC;


-- Joins - # of trips by station, + average trip time

SELECT
  s.name AS station_name,
  COUNT(*) AS num_trips,
  ROUND(AVG(t.tripduration) / 60, 2) AS avg_trip_minutes
FROM
  `bigquery-public-data.new_york.citibike_trips` t
JOIN
  `bigquery-public-data.new_york.citibike_stations` s
  ON t.start_station_name = s.name -- no PK/FK! Denormalized!
GROUP BY
  s.name
ORDER BY
  num_trips DESC
LIMIT 20;




Next Queries (heavy on window functions):------------------------------

(Chicago Taxi Trips)

-- Busiest hours of the day + average fare
SELECT
  EXTRACT(HOUR FROM trip_start_timestamp) AS hour_of_day,
  COUNT(*) AS num_trips,
  ROUND(AVG(fare), 2) AS avg_fare
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE
  fare > 0
GROUP BY
  hour_of_day
ORDER BY
  num_trips DESC;

-- Rank Companies by Revenue (top 10) (SKIP THIS ONE PROBS)
SELECT
  company,
  COUNT(*) AS num_trips,
  ROUND(SUM(fare), 2) AS total_revenue,
  ROUND(AVG(tips), 2) AS avg_tip
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE

GROUP BY

LIMIT 10;

-- Window function that exceeds peak use limit
-- Tries to get a running total of all trips, grouping by start time
SELECT
  trip_start_timestamp,
  COUNT(*) AS trips_at_this_timestamp,
  SUM(COUNT(*)) OVER (ORDER BY trip_start_timestamp) AS running_total_trips
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE
  fare > 0
GROUP BY
  trip_start_timestamp
ORDER BY
  trip_start_timestamp
LIMIT 20;

-- The fix: Only do this for a single year!
--This heavily narrows the results, and we can complete the query
SELECT
  trip_start_timestamp,
  COUNT(*) AS trips_at_this_timestamp,
  SUM(COUNT(*)) OVER (ORDER BY trip_start_timestamp) AS running_total_trips
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE
  fare > 0
  AND EXTRACT(YEAR FROM trip_start_timestamp) = 2023
GROUP BY
  trip_start_timestamp
ORDER BY
  trip_start_timestamp
LIMIT 20;

Use RANK with a window function to find the company with the most trip volume each month
-- Find the company with the most trips each month in 2023 (RANK window function)
SELECT
  company AS top_ranked_company,
  EXTRACT(MONTH FROM trip_start_timestamp) AS month,
  COUNT(*) AS num_trips
FROM
  `bigquery-public-data.chicago_taxi_trips.taxi_trips`
WHERE
  company IS NOT NULL
  AND fare > 0
  AND EXTRACT(YEAR FROM trip_start_timestamp) = 2023
GROUP BY
  company, month
-- Qualify is an easy way to filter results based on a window function
-- Rank() is our window function here, giving us the company with the most trips each month
QUALIFY RANK() OVER (PARTITION BY month ORDER BY COUNT(*) DESC) = 1
-- We will only get records that have a rank of 1 within their month, meaning the single highest trip-count company for that month
ORDER BY
  month;
